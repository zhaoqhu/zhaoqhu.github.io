高并发的设计

LVS/LB 在更底层做分发，Nginx/网关做七层治理，真正“分布式”的主体实现在应用层（代码/架构）和数据层（存储/缓存/MQ）。

你要做的事情 主要落在哪（常见做法） 典型技术
扛大流量、四层负载均衡 网络层/传输层（L4） LVS、HAProxy(tcp)、硬件LB
接入层限流/路由/SSL卸载/七层转发 应用层（L7） Nginx、API Gateway（Kong/Apisix…）
Web服务器水平扩容 应用层 PHP-FPM横向扩容、多实例
分布式能力（缓存/队列/数据库拆分/微服务） 应用层 + 数据层 Redis/Memcached、RabbitMQ/Kafka、MySQL分库分表/读写分离、微服务RPC
远程机器间通信 传输层 TCP/UDP（由框架/中间件使用）

熔断（Go/PHP应用层）

调用支付接口连续失败5次，暂停调用30秒，防止下游故障拖死整个系统。

$key = 'pay:fail:count';
$threshold = 5;      // 连续失败 5 次
$coolDown = 30;      // 熔断 30 秒

$redis = new Redis();
$redis->connect('127.0.0.1', 6379);

// 检查是否熔断
if ($redis->get('pay:circuit:open')) {
    echo json_encode(['code' => 500, 'msg' => '支付服务暂不可用']);
    return;
}

try {
    $result = call_payment_api();

    if ($result['success']) {
        $redis->del($key);
        echo json_encode(['code' => 0]);
    } else {
        throw new Exception();
    }
} catch (Exception $e) {
    $fails = $redis->incr($key);
    if ($fails >= $threshold) {
        $redis->setex('pay:circuit:open', $coolDown, 1);
    }
    echo json_encode(['code' => 500]);
}

        stub_status;
        allow 127.0.0.1;
        deny all;
    }

# ==============================
# 全局配置
# ==============================
user nginx;
worker_processes auto;
pid /run/nginx.pid;

events {
    worker_connections 10240;
    use epoll;
    multi_accept on;
}

# ==============================
# HTTP 核心配置
# ==============================
http {
    include       mime.types;
    default_type  application/octet-stream;

    # ---------- 日志 ----------
    log_format main
        '$remote_addr - $remote_user [$time_local] "$request" '
        '$status $body_bytes_sent "$http_referer" '
        '"$http_user_agent" "$http_x_forwarded_for"';

    access_log /var/log/nginx/access.log main buffer=32k flush=5s;
    error_log  /var/log/nginx/error.log warn;

    # ---------- 基础优化 ----------
    sendfile           on;
    tcp_nopush         on;
    tcp_nodelay        on;
    keepalive_timeout  65;
    types_hash_max_size 2048;

    # ---------- 安全 ----------
    server_tokens off;

    # ---------- Gzip ----------
    gzip on;
    gzip_comp_level 5;
    gzip_min_length 1k;
    gzip_types
        text/plain
        text/css
        application/json
        application/javascript
        text/xml
        application/xml
        application/xml+rss
        text/javascript;

    # ---------- 限速（防刷） ----------
    limit_req_zone $binary_remote_addr zone=req_limit:10m rate=10r/s;

    # ==========================
    # 站点：HTTP（强制 HTTPS）
    # ==========================
    server {
        listen 80;
        server_name example.com www.example.com;

        # HTTP 跳转 HTTPS
        return 301 https://$host$request_uri;
    }

    # ==========================
    # 站点：HTTPS（生产用）
    # ==========================
    server {
        listen 443 ssl http2;
        server_name example.com www.example.com;

        root /var/www/example.com;
        index index.html index.htm index.php;

        # ---------- SSL ----------
        ssl_certificate     /etc/nginx/ssl/example.com/fullchain.pem;
        ssl_certificate_key /etc/nginx/ssl/example.com/privkey.pem;
        ssl_protocols TLSv1.2 TLSv1.3;
        ssl_ciphers HIGH:!aNULL:!MD5;
        ssl_prefer_server_ciphers on;

        # ---------- 安全头 ----------
        add_header X-Frame-Options DENY;
        add_header X-Content-Type-Options nosniff;
        add_header X-XSS-Protection "1; mode=block";

        # ---------- 限速 ----------
        limit_req zone=req_limit burst=20 nodelay;

        # ---------- 静态资源缓存 ----------
        location ~* \.(js|css|png|jpg|jpeg|gif|ico|svg)$ {
            expires 30d;
            add_header Cache-Control "public, immutable";
            access_log off;
        }

        # ---------- 主路由 ----------
        location / {
            try_files $uri $uri/ /index.html;
        }

        # ---------- PHP（如需要） ----------
        # location ~ \.php$ {
        #     fastcgi_pass unix:/run/php/php-fpm.sock;
        #     fastcgi_index index.php;
        #     include fastcgi_params;
        # }

        # ---------- 禁止隐藏文件 ----------
        location ~ /\. {
            deny all;
            access_log off;
            log_not_found off;
        }
    }

    # ==========================
    # Nginx 状态监控（仅本地）
    # ==========================
    server {
        listen 127.0.0.1:80;
        server_name localhost;

        location /nginx_status {
            stub_status;
            allow 127.0.0.1;
            deny all;
        }
    }
}
