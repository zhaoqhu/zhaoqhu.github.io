消息队列

生产者将消息发往交换机，交换机依据路由规则和绑定关系将消息投递到指定队列；队列持久化存储消息，消费者监听队列获取消息，处理完成后发送 ACK，RabbitMQ 确认后删除消息 —— 形成一条完整、可靠、可控的消息流转链路。

生产者

✅ 如何确保“几乎 100%”不丢？—— 5 层防护详解

层级 机制 解决的问题
1️⃣ 幂等控制 Redis SET NX 防止并发压测导致重复发送
2️⃣ 持久化 delivery_mode=2 + durable=true 防止 Broker 重启丢消息
3️⃣ Publisher Confirms ack/nack 确认消息被 Broker 接收并持久化
4️⃣ Mandatory + Return 路由失败返回 防止消息因路由错误被静默丢弃
5️⃣ 失败回滚 删除 Redis 标记 允许重试，避免永久阻塞

生产环境中生产者的标准答案

没有一个 basic_publish()能保证消息入队，但“持久化 + confirms + mandatory + return + 幂等 + 重试”的组合拳，可以让消息丢失概率低于 0.001%，达到工业级可靠性

<?php
require_once __DIR__ . '/vendor/autoload.php';

use PhpAmqpLib\Connection\AMQPStreamConnection;
use PhpAmqpLib\Message\AMQPMessage;
use PhpAmqpLib\Exception\AMQPTimeoutException;

class ReliableRabbitMQProducer
{
    private $connection;
    private $redis; // 用于幂等控制

    public function __construct()
    {
        // 1. 建立 RabbitMQ 连接（进程内复用）
        $this->connection = new AMQPStreamConnection(
            'rabbitmq', 5672, 'guest', 'guest',
            '/', false, 'plain', null, 'en_US',
            3.0, 3.0, null, false, 60
        );

        // 2. 初始化 Redis（用于幂等）
        $this->redis = new Redis();
        $this->redis->connect('redis', 6379);
    }

    /**
     * 可靠发送消息（多层防护）
     * @param string $orderId 业务唯一ID
     * @param array $orderData 订单数据
     * @return bool 是否成功
     */
    public function sendOrderCreated(string $orderId, array $orderData): bool
    {
        $lockKey = "sent:order:{$orderId}";

        // === 第1层：幂等控制（防并发压测）===
        $isFirstSend = $this->redis->set($lockKey, 1, ['nx', 'ex' => 86400]);
        if (!$isFirstSend) {
            error_log("Duplicate request ignored: {$orderId}");
            return true; // 幂等返回成功
        }

        $channel = null;
        try {
            $channel = $this->connection->channel();

            // === 第2层：声明资源（幂等操作）===
            $exchange = 'orders';
            $queue = 'order.created';
            $routingKey = 'order.created';

            $channel->exchange_declare($exchange, 'direct', false, true, false);
            $channel->queue_declare($queue, false, true, false, false);
            $channel->queue_bind($queue, $exchange, $routingKey);

            // === 第3层：构造持久化消息 + mandatory ===
            $messageId = 'order:' . $orderId . ':' . str_replace('.', '', microtime(true));
            $msg = new AMQPMessage(
                json_encode([
                    'message_id' => $messageId,
                    'order_id'   => $orderId,
                    'data'       => $orderData,
                    'timestamp'  => time()
                ]),
                [
                    'delivery_mode' => AMQPMessage::DELIVERY_MODE_PERSISTENT, // 持久化
                    'content_type'  => 'application/json',
                    'message_id'    => $messageId,
                    'mandatory'     => true  // 关键：路由失败返回给生产者
                ]
            );

            // === 第4层：开启 Publisher Confirms ===
            $channel->confirm_select();

            $acked = false;
            $nacked = false;
            $returned = false;

            // 设置 ack/nack 回调
            $channel->set_ack_handler(function (AMQPMessage $message) use (&$acked, $messageId) {
                if ($message->get('message_id') === $messageId) {
                    $acked = true;
                    error_log("✅ [CONFIRMED] Message acked: {$messageId}");
                }
            });

            $channel->set_nack_handler(function (AMQPMessage $message) use (&$nacked, $messageId) {
                if ($message->get('message_id') === $messageId) {
                    $nacked = true;
                    error_log("❌ [FAILED] Message nacked: {$messageId}");
                }
            });

            // 设置 return 回调（路由失败）
            $channel->set_return_listener(function ($replyCode, $replyText, $exchange, $routingKey, AMQPMessage $message) use (&$returned, $messageId) {
                $returned = true;
                error_log("🚨 [RETURNED] Message returned: {$replyCode} {$replyText}, msg: {$messageId}");
                // 可记录到数据库，人工处理
            });

            // === 第5层：发送消息 ===
            $channel->basic_publish($msg, $exchange, $routingKey);

            // === 第6层：等待 Broker 确认（同步阻塞，最多5秒）===
            $channel->wait_for_pending_acks_returns(5.0);

            // === 第7层：判断结果 ===
            if ($nacked) {
                throw new Exception("Broker nack'd the message (e.g., queue full)");
            }
            if ($returned) {
                throw new Exception("Message was returned (routing failed)");
            }
            if (!$acked) {
                throw new Exception("No ack received (network issue or broker down)");
            }

            // ✅ 成功：消息已入队并持久化
            error_log("🎉 Message successfully persisted to queue: {$messageId}");
            return true;

        } catch (Exception $e) {
            // === 第8层：失败回滚幂等标记（允许重试）===
            $this->redis->del($lockKey);
            error_log("❌ Send failed: {$e->getMessage()} | Order: {$orderId}");
            return false;
        } finally {
            // 清理资源
            if ($channel && $channel->is_open()) {
                $channel->close();
            }
        }
    }
}

    public function sendOrderCreated(string $orderId, array $data): bool
{
  $lockKey = "sent:order:{$orderId}";
  $tempKey = "temp:order:{$orderId}"; // 可选：用临时标记

  // 方案 A：直接 SET NX（简单）
  $isFirst = $this->redis->set($lockKey, 1, ['nx', 'ex' => 3600]);

  if (!$isFirst) {
      return true; // 幂等
  }

  // 发送消息
  try {
      $this->doPublish($orderId, $data);
      return true; // 成功
  } catch (Exception $e) {
      // 关键：发送失败，清除标记！
      $this->redis->del($lockKey);
      error_log("Publish failed, rollback mark: {$orderId}");
      return false;
  }
}

    🛡️ RabbitMQ 消费者防重复消费与幂等处理完整方案

核心思想：RabbitMQ 只保证“至少一次”或“至多一次”交付，不保证“恰好一次”。因此，消费者必须具备识别重复消息、安全忽略重复、保证业务幂等的能力，否则将导致资损、数据错乱等严重问题。

一、为什么必须处理重复消费？

原因 说明
🔁 网络抖动或处理超时 RabbitMQ 未收到 ACK，重投消息
💥 消费者崩溃 业务已执行但未 ACK，重启后消息重发
🧱 手动 ACK 遗漏 代码 Bug 导致忘记 basic_ack()
🔄 集群 HA 切换 镜像队列主从切换可能触发重发
📤 生产者重试 网络问题导致同一条消息被发多次

👉 结论：重复消费是分布式系统中的常态，必须设计幂等机制应对。

二、幂等性：消费者的生命线

定义：无论同一条消息被消费多少次，系统的业务状态和最终结果都与只消费一次相同。

✅ 幂等不是“可选项”，而是生产环境的必选项。

三、五大幂等策略（按推荐度排序）

方案 原理 适用场景 优点 缺点
1️⃣ 基于 message_id + Redis/NX 消费前用 SET message_id NX EX 判断是否已处理 通用场景，尤其异步消息 ✅ 简单高效
✅ 分布式友好
✅ 可设置过期		 ❌ 依赖 Redis 可用性
2️⃣ 基于业务主键（如 order_id） 用业务唯一 ID 判断是否已处理 有明确业务标识的场景 ✅ 无需全局 ID
✅ 贴近业务逻辑		 ❌ 同一业务可能多次合法更新（如状态变更）
3️⃣ 数据库唯一约束 插入时依赖 UNIQUE KEY 报错 数据入库、用户注册等 ✅ 强一致性
✅ 不依赖外部缓存		 ❌ 性能差
❌ 需捕获异常
4️⃣ 状态机控制 业务状态只能单向流转（如：待支付 → 已支付） 有严格状态流转的业务 ✅ 天然防重
✅ 无额外存储		 ❌ 不适用于幂等更新（如多次加积分）
5️⃣ 本地文件或内存标记 单机脚本可用 flock() 或数组缓存 单机、低频场景 ✅ 简单 ❌ 不支持分布式
❌ 重启丢失

✅ 推荐组合：

  • 常规业务：方案 1（message_id + Redis）
  • 核心交易：方案 1 + 方案 3（数据库唯一键）双重保险
  • 状态驱动：方案 4（状态机）为主，辅以方案 1

四、消费者开发十大关键实践（整合可靠性与可维护性）

类别 实践要点 说明
🔐 1. 手动 ACK 禁用 auto_ack=true，业务完成后 basic_ack() 防止未处理完就确认，避免消息丢失
⚖️ 2. 公平分发（QoS） basic_qos(null, 1, null) 控制未 ACK 消息数 避免某消费者过载，提升负载均衡
🧼 3. 优雅关闭 捕获 SIGTERM，停止 wait() 并等待当前消息处理完 防止强制退出导致消息重投
🗑️ 4. 死信队列（DLX） 配置 DLX 处理重试超限的消息 避免无限重入队，便于问题排查
🕒 5. 控制处理时长 避免阻塞操作，设置 read_write_timeout 防止心跳超时导致连接断开
🔁 6. 重试策略 失败后 nack(true) 重试有限次数（如 3 次） 防死循环，结合 DLX 使用
📊 7. 日志与监控 记录 message_id、处理结果、重试次数 便于追踪问题，支持对账
🆔 8. 唯一标识传递 消息体必须包含 message_id（生产者生成） 幂等去重的依据
🧩 9. 幂等优先于重试 先判断是否已处理，再决定是否执行业务 减少无效重试，提升性能
🧪 10. 压测验证 模拟网络抖动、消费者崩溃，验证去重有效性 确保方案在生产环境可靠

六、总结：构建“零资损”消费者体系的四大支柱

支柱 内涵 落地动作
1️⃣ 认知到位 接受“重复消费是常态” 设计时默认消息可能重复
2️⃣ 幂等优先 业务层具备去重能力 采用 message_id + Redis 等方案
3️⃣ 机制健全 ACK、QoS、DLX、监控全覆盖 避免技术性重复（如未 ACK）
4️⃣ 验证充分 压测 + 对账 + 监控 确保方案在真实场景中可靠

✅ 最终建议

  • 不要依赖 RabbitMQ 保证“只消费一次” —— 它没有这个语义。
  • 不要把“没报错”当成“成功” —— 必须 ACK 才算完成。
  • 不要把“重试”当“容错” —— 无幂等的重试是灾难。
  • 把“幂等”当作和“业务逻辑”同等重要的模块来设计。

💡 金句总结：

在消息驱动的系统中，消费者不是“执行者”，而是“决策者”——它必须判断：这条消息我以前见过吗？我现在该处理吗？处理错了能回滚吗？只有想清楚这三个问题，系统才能真正可靠。

<?php
namespace App\Consumers;

use PhpAmqpLib\Connection\AMQPStreamConnection;
use PhpAmqpLib\Message\AMQPMessage;
use PhpAmqpLib\Exception\AMQPTimeoutException;
use Redis;
use Psr\Log\LoggerInterface;

abstract class BaseConsumer
{
    protected $connection;
    protected $channel;
    protected $redis;
    protected $logger;
    protected $config;
    
    // 信号控制
    protected $shouldStop = false;
    protected $currentMessage = null;
    protected $currentStartTime = 0;

    public function __construct(array $config, LoggerInterface $logger)
    {
        $this->config = $config;
        $this->logger = $logger;
        $this->setupSignalHandlers();
    }

    /**
     * 初始化连接和资源
     */
    public function initialize(): void
    {
        // 初始化 Redis
        $this->initializeRedis();
        
        // 初始化 RabbitMQ 连接
        $this->initializeRabbitMQ();
    }

    /**
     * 初始化 Redis
     */
    protected function initializeRedis(): void
    {
        try {
            $this->redis = new Redis();
            $redisConfig = $this->config['redis'];
            
            if (!$this->redis->connect(
                $redisConfig['host'], 
                $redisConfig['port'], 
                $redisConfig['timeout']
            )) {
                throw new \Exception("Redis connection failed");
            }
            
            if (!empty($redisConfig['password'])) {
                $this->redis->auth($redisConfig['password']);
            }
            
            $this->redis->select($redisConfig['database']);
            
            $this->logger->info("Redis connected successfully");
        } catch (\Exception $e) {
            $this->logger->error("Redis initialization failed: " . $e->getMessage());
            throw $e;
        }
    }

    /**
     * 初始化 RabbitMQ 连接
     */
    protected function initializeRabbitMQ(): void
    {
        try {
            $rabbitConfig = $this->config['rabbitmq'];
            
            $this->connection = new AMQPStreamConnection(
                $rabbitConfig['host'],
                $rabbitConfig['port'],
                $rabbitConfig['username'],
                $rabbitConfig['password'],
                $rabbitConfig['vhost'],
                false,
                'AMQPLAIN',
                null,
                'en_US',
                $rabbitConfig['read_write_timeout'],
                $rabbitConfig['read_write_timeout'],
                null,
                false,
                $rabbitConfig['heartbeat']
            );

            $this->channel = $this->connection->channel();
            
            // 设置 QoS（公平分发）
            $this->channel->basic_qos(null, $this->getPrefetchCount(), null);
            
            $this->logger->info("RabbitMQ connected successfully");
        } catch (\Exception $e) {
            $this->logger->error("RabbitMQ initialization failed: " . $e->getMessage());
            throw $e;
        }
    }

    /**
     * 声明队列和绑定
     */
    protected function declareQueue(string $queueName): void
    {
        $queueConfig = $this->config['queues'][$queueName];
        
        // 声明死信交换机和队列（如果启用）
        if ($this->config['dead_letter']['enabled']) {
            $this->declareDeadLetter($queueName);
        }

        // 声明主队列
        $args = [];
        if ($this->config['dead_letter']['enabled']) {
            $args['x-dead-letter-exchange'] = $this->config['dead_letter']['exchange'];
            $args['x-dead-letter-routing-key'] = $this->config['dead_letter']['queue'];
            $args['x-message-ttl'] = $this->config['dead_letter']['ttl'];
        }

        $this->channel->queue_declare(
            $queueConfig['name'],
            false, // passive
            $queueConfig['durable'], // durable
            false, // exclusive
            false, // auto_delete
            false, // nowait
            $args // arguments
        );

        // 声明交换机
        $this->channel->exchange_declare(
            $queueConfig['exchange'],
            'direct', // type
            false, // passive
            $queueConfig['durable'], // durable
            false // auto_delete
        );

        // 绑定队列到交换机
        $this->channel->queue_bind(
            $queueConfig['name'],
            $queueConfig['exchange'],
            $queueConfig['routing_key']
        );

        $this->logger->info("Queue {$queueName} declared and bound successfully");
    }

    /**
     * 声明死信队列
     */
    protected function declareDeadLetter(string $queueName): void
    {
        $dlxConfig = $this->config['dead_letter'];
        $queueConfig = $this->config['queues'][$queueName];

        // 声明死信交换机
        $this->channel->exchange_declare(
            $dlxConfig['exchange'],
            'direct',
            false,
            true,
            false
        );

        // 声明死信队列
        $this->channel->queue_declare(
            $dlxConfig['queue'] . '.' . $queueName,
            false,
            true,
            false,
            false
        );

        // 绑定死信队列
        $this->channel->queue_bind(
            $dlxConfig['queue'] . '.' . $queueName,
            $dlxConfig['exchange'],
            $queueConfig['routing_key']
        );
    }

    /**
     * 消息处理函数（子类必须实现）
     */
    abstract protected function processMessage(array $data, array $headers): bool;

    /**
     * 获取消息去重键
     */
    protected function getDeduplicationKey(AMQPMessage $message): string
    {
        $data = json_decode($message->getBody(), true);
        return $data['message_id'] ?? $message->getDeliveryTag() . '_' . md5($message->getBody());
    }

    /**
     * 检查消息是否已处理（幂等性）
     */
    protected function isMessageProcessed(string $deduplicationKey): bool
    {
        $key = "processed:message:{$deduplicationKey}";
        return $this->redis->exists($key);
    }

    /**
     * 标记消息已处理
     */
    protected function markMessageProcessed(string $deduplicationKey, int $ttl = 86400): void
    {
        $key = "processed:message:{$deduplicationKey}";
        $this->redis->setex($key, $ttl, 1);
    }

    /**
     * 消息消费回调
     */
    public function onMessage(AMQPMessage $message): void
    {
        $this->currentMessage = $message;
        $this->currentStartTime = microtime(true);

        try {
            // 提取消息内容
            $data = json_decode($message->getBody(), true);
            if (json_last_error() !== JSON_ERROR_NONE) {
                throw new \Exception("Invalid JSON: " . json_last_error_msg());
            }

            $headers = $message->get_properties();
            $deduplicationKey = $this->getDeduplicationKey($message);

            $this->logger->info("Processing message", [
                'message_id' => $data['message_id'] ?? 'unknown',
                'delivery_tag' => $message->getDeliveryTag(),
                'redelivered' => $message->isRedelivered()
            ]);

            // 幂等性检查
            if ($this->isMessageProcessed($deduplicationKey)) {
                $this->logger->info("Duplicate message detected, skipping", [
                    'message_id' => $data['message_id'] ?? 'unknown'
                ]);
                $message->ack();
                return;
            }

            // 业务处理
            $success = $this->processMessage($data, $headers);

            if ($success) {
                // 标记消息已处理
                $this->markMessageProcessed($deduplicationKey);
                
                // 确认消息
                $message->ack();
                
                $processingTime = round((microtime(true) - $this->currentStartTime) * 1000, 2);
                $this->logger->info("Message processed successfully", [
                    'message_id' => $data['message_id'] ?? 'unknown',
                    'processing_time_ms' => $processingTime
                ]);
            } else {
                throw new \Exception("Business processing failed");
            }

        } catch (\Exception $e) {
            $this->handleProcessingFailure($message, $e);
        } finally {
            $this->currentMessage = null;
            $this->currentStartTime = 0;
        }
    }

    /**
     * 处理处理失败
     */
    protected function handleProcessingFailure(AMQPMessage $message, \Exception $e): void
    {
        $data = json_decode($message->getBody(), true);
        $messageId = $data['message_id'] ?? 'unknown';
        
        $processingTime = round((microtime(true) - $this->currentStartTime) * 1000, 2);
        
        $this->logger->error("Message processing failed", [
            'message_id' => $messageId,
            'error' => $e->getMessage(),
            'processing_time_ms' => $processingTime,
            'delivery_tag' => $message->getDeliveryTag(),
            'redelivered' => $message->isRedelivered()
        ]);

        // 根据重试次数决定是否重试或拒绝
        $retryCount = $this->getCurrentRetryCount($message);
        $maxRetries = $this->getMaxRetryCount();

        if ($retryCount < $maxRetries) {
            // 重试：重新入队
            $this->logger->info("Retrying message", [
                'message_id' => $messageId,
                'retry_count' => $retryCount + 1,
                'max_retries' => $maxRetries
            ]);
            
            // 延迟重试（通过 TTL 实现）
            $message->nack(false); // 不重新入队，等待 TTL 后进入死信或重新发布
        } else {
            // 达到最大重试次数，拒绝消息（进入死信队列）
            $this->logger->warning("Max retries exceeded, sending to dead letter", [
                'message_id' => $messageId,
                'retry_count' => $retryCount
            ]);
            
            $message->nack(false); // 拒绝，不重新入队
        }
    }

    /**
     * 获取当前重试次数
     */
    protected function getCurrentRetryCount(AMQPMessage $message): int
    {
        $headers = $message->get_properties();
        $retryCount = $headers['application_headers']['x-retry-count'][1] ?? 0;
        return (int)$retryCount;
    }

    /**
     * 获取最大重试次数
     */
    protected function getMaxRetryCount(): int
    {
        return $this->config['queues']['order.created']['retry_count'] ?? 3;
    }

    /**
     * 获取预取计数
     */
    protected function getPrefetchCount(): int
    {
        return $this->config['queues']['order.created']['prefetch'] ?? 1;
    }

    /**
     * 设置信号处理器（优雅关闭）
     */
    protected function setupSignalHandlers(): void
    {
        pcntl_signal(SIGTERM, [$this, 'signalHandler']);
        pcntl_signal(SIGINT, [$this, 'signalHandler']);
        pcntl_signal(SIGHUP, [$this, 'signalHandler']);
    }

    /**
     * 信号处理
     */
    public function signalHandler(int $signal): void
    {
        $this->logger->info("Received signal {$signal}, shutting down gracefully...");
        $this->shouldStop = true;
    }

    /**
     * 开始消费
     */
    public function startConsuming(string $queueName): void
    {
        $this->declareQueue($queueName);

        $this->logger->info("Starting consumer for queue: {$queueName}");

        $callback = function (AMQPMessage $message) {
            // 检查是否需要停止
            pcntl_signal_dispatch();
            if ($this->shouldStop) {
                $this->logger->info("Stopping consumer due to shutdown signal");
                throw new \Exception("Consumer shutdown requested");
            }

            $this->onMessage($message);
        };

        $queueConfig = $this->config['queues'][$queueName];
        
        $this->channel->basic_consume(
            $queueConfig['name'],
            '', // consumer tag
            false, // no local
            false, // no ack (we do manual ack)
            false, // exclusive
            false, // no wait
            $callback
        );

        // 主循环
        try {
            while (count($this->channel->callbacks) && !$this->shouldStop) {
                $this->channel->wait(null, false, 1); // 1秒超时，便于检查停止信号
                pcntl_signal_dispatch();
            }
        } catch (\Exception $e) {
            $this->logger->info("Consumer stopped: " . $e->getMessage());
        }

        $this->logger->info("Consumer stopped gracefully");
    }

    /**
     * 关闭连接
     */
    public function close(): void
    {
        try {
            if ($this->channel && $this->channel->is_open()) {
                $this->channel->close();
            }
            if ($this->connection && $this->connection->isConnected()) {
                $this->connection->close();
            }
            if ($this->redis) {
                $this->redis->close();
            }
            $this->logger->info("All connections closed");
        } catch (\Exception $e) {
            $this->logger->error("Error closing connections: " . $e->getMessage());
        }
    }

    public function __destruct()
    {
        $this->close();
    }
}

<?php
require_once __DIR__ . '/../vendor/autoload.php';

use App\Consumers\OrderCreatedConsumer;
use Monolog\Logger;
use Monolog\Handler\RotatingFileHandler;
use Monolog\Formatter\LineFormatter;

// 加载配置
$config = require_once __DIR__ . '/../config/rabbitmq.php';

// 初始化日志
$logger = new Logger('rabbitmq_consumer');

// 文件处理器（按天轮转，保留30天）
$fileHandler = new RotatingFileHandler(
    __DIR__ . '/../logs/consumer.log',
    30, // 保留天数
    Logger::INFO
);

$formatter = new LineFormatter(
    "[%datetime%] %channel%.%level_name%: %message% %context% %extra%\n",
    "Y-m-d H:i:s.u"
);
$fileHandler->setFormatter($formatter);
$logger->pushHandler($fileHandler);

try {
    // 创建消费者实例
    $consumer = new OrderCreatedConsumer($config, $logger);
    
    // 初始化连接
    $consumer->initialize();
    
    // 开始消费（优雅关闭支持）
    $consumer->startConsuming('order.created');
    
} catch (\Exception $e) {
    $logger->critical("Consumer failed to start: " . $e->getMessage(), [
        'exception' => $e->getTraceAsString()
    ]);
    exit(1);
}
