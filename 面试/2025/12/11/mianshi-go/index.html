package main

import (
	"context"
	"fmt"
	"net/http"
	"strings"
	"sync"
	"time"

	"github.com/gin-gonic/gin" // 用 Gin 模拟 HTTP 服务（需先安装：go get -u github.com/gin-gonic/gin）
)

// 自定义键类型（避免键冲突）
type ctxKey string

const (
	KeyTraceID ctxKey = "traceID" // 追踪 ID 键
	KeyUserID  ctxKey = "userID"  // 用户 ID 键
)

// 中间件：注入 TraceID 和用户身份信息到 Context
func metadataMiddleware() gin.HandlerFunc {
	return func(c *gin.Context) {
		// 1. 生成 TraceID（模拟 UUID，实际可用 uuid 库）
		traceID := fmt.Sprintf("trace-%d", time.Now().UnixNano())

		// 2. 从 Header 获取 Token 并解析 UserID（模拟认证逻辑）
		authHeader := c.GetHeader("Authorization")
		userID := 0
		if strings.HasPrefix(authHeader, "Bearer ") {
			// 模拟解析 Token 得到 UserID=1001（实际需用 JWT 库解析）
			userID = 1001
		}

		// 3. 将元数据存入 Context（链式传递）
		ctx := context.WithValue(c.Request.Context(), KeyTraceID, traceID)
		ctx = context.WithValue(ctx, KeyUserID, userID)
		c.Request = c.Request.WithContext(ctx) // 更新请求的 Context

		c.Next() // 继续处理请求
	}
}

// 处理函数：从 Context 获取元数据，并启动子协程传递元数据
func handleRequest(c *gin.Context) {
	ctx := c.Request.Context()

	// 1. 从 Context 获取元数据（TraceID、UserID）
	traceID := ctx.Value(KeyTraceID).(string)
	userID := ctx.Value(KeyUserID).(int)

	fmt.Printf("[主协程] TraceID: %s, UserID: %d\n", traceID, userID)

	// 2. 启动子协程，传递 Context（元数据随之传递）
	var wg sync.WaitGroup
	wg.Add(1)
	go func(ctx context.Context) {
		defer wg.Done()
		// 子协程从 Context 获取元数据
		subTraceID := ctx.Value(KeyTraceID).(string)
		subUserID := ctx.Value(KeyUserID).(int)
		fmt.Printf("[子协程] TraceID: %s, UserID: %d\n", subTraceID, subUserID)
	}(ctx) // 传递当前请求的 Context

	wg.Wait() // 等待子协程完成
	c.JSON(http.StatusOK, gin.H{"status": "success"})
}

func main() {
	r := gin.Default()
	r.Use(metadataMiddleware())   // 全局启用元数据中间件
	r.GET("/demo", handleRequest) // 注册接口

	fmt.Println("服务启动：http://localhost:8080/demo")
	fmt.Println("请求示例：curl -H 'Authorization: Bearer token' http://localhost:8080/demo")
	r.Run(":8080")
}

package main

import (
	"context"
	"fmt"
	"sync"
	"time"
)

func main() {
	var wg sync.WaitGroup
	ctx, cancel := context.WithTimeout(context.Background(), 5*time.Second) // 5秒超时
	defer cancel()

	wg.Add(1)
	go func() {
		defer wg.Done()
		select {
		case <-time.After(10 * time.Second): // 模拟 10 秒任务（超时）
			fmt.Println("任务完成")
		case <-ctx.Done(): // 监听超时信号
			fmt.Println("任务超时:", ctx.Err())
		}
	}()

	wg.Wait() // 等待协程退出（或被取消后退出）
	fmt.Println("主程序结束")
}

3. golang中怎么退出一个协程？

父协程通过Context发送Cancel取消信号，子协程通过select监听Done信号退出。 父协程通过close关闭channel，子协程通过监听channel关闭信息退出。

package main

import (
	"context"
	"fmt"
	"sync"
	"time"
)

// 子协程函数：监听 Context 取消信号退出
func workerWithContext(ctx context.Context, id int, wg *sync.WaitGroup) {
	defer wg.Done() // 确保协程退出时通知 WaitGroup
	fmt.Printf("协程 %d 启动\n", id)

	for {
		select {
		case <-ctx.Done(): // 监听取消信号（Context 被取消时，该通道关闭）
			fmt.Printf("协程 %d 收到退出信号，原因：%v\n", id, ctx.Err())
			return // 主动退出协程
		default:
			// 模拟业务逻辑（如处理任务）
			fmt.Printf("协程 %d 工作中...（模拟任务）\n", id)
			time.Sleep(500 * time.Millisecond) // 模拟耗时操作
		}
	}
}

func main() {
	var wg sync.WaitGroup
	// 1. 创建可取消的 Context（父 Context 为 context.Background()）
	ctx, cancel := context.WithCancel(context.Background())
	defer cancel() // 确保主程序退出时取消 Context，避免资源泄漏

	// 2. 启动 3 个子协程（传递 Context 和 WaitGroup）
	for i := 1; i <= 3; i++ {
		wg.Add(1)
		go workerWithContext(ctx, i, &wg)
	}

	// 3. 主协程运行 2 秒后发送取消信号
	time.Sleep(2 * time.Second)
	fmt.Println("\n===== 主协程发送取消信号 =====")
	cancel() // 关闭 ctx.Done() 通道，所有子协程收到退出信号

	// 4. 等待所有子协程退出
	wg.Wait()
	fmt.Println("===== 所有子协程已退出，主程序结束 =====")
}

package main

import (
	"fmt"
	"math/rand"
	"time"
)

func taskGen(ch chan int, name string) {
	for i := 0; i < 3; i++ {
		time.Sleep(time.Duration(rand.Intn(1000)) * time.Millisecond) // 随机延迟
		ch <- rand.Intn(100)                                          // 发送随机任务数据
		fmt.Printf("%s 发送任务: %d\n", name, i)
	}
	close(ch) // 任务发送完毕，关闭通道
}

func main() {
	ch1 := make(chan int, 5)
	ch2 := make(chan int, 5)
	ch3 := make(chan int, 5)

	// 启动3个任务生成器
	go taskGen(ch1, "任务源1")
	go taskGen(ch2, "任务源2")
	go taskGen(ch3, "任务源3")

	// 用 select 随机选择就绪通道处理任务
	for {
		select {
		case data, ok := <-ch1:
			// 通道关闭后设为 nil，避免下次监听
			if !ok {
				ch1 = nil
			} else {
				fmt.Printf("处理 ch1 任务: %d\n", data)
			}
		case data, ok := <-ch2:
			if !ok {
				ch2 = nil
			} else {
				fmt.Printf("处理 ch2 任务: %d\n", data)
			}
		case data, ok := <-ch3:
			if !ok {
				ch3 = nil
			} else {
				fmt.Printf("处理 ch3 任务: %d\n", data)
			}
		}
		// 所有通道都关闭时退出
		if ch1 == nil && ch2 == nil && ch3 == nil {
			break
		}
	}
	fmt.Println("所有任务处理完毕")
}

5. golang中多协程goroutine操作map数据类型，会怎么样？

当多个 goroutine 同时对 map 进行读写操作（尤其是写操作）时，会触发数据竞争（Data Race），导致未定义行为，包括但不限于： 程序崩溃（panic）（如 fatal error: concurrent map read and map write）； 解决方法是使用sync.Mutex排他锁或者使用sync.RWMutex。

func main() {
	m := make(map[int]int)
	var wg sync.WaitGroup
	var rwmu sync.RWMutex

	// 启动 2 个 goroutine 同时写 map
	for i := 0; i < 2; i++ {
		wg.Add(1)
		go func(idx int) {
			defer wg.Done()
			for j := 0; j < 1000; j++ {
				rwmu.Lock()         // 加锁
				m[j] = idx*1000 + j // 写操作（并发冲突！）
				rwmu.Unlock()       // 解锁
			}
		}(i)
	}

	wg.Wait()
	fmt.Println("map 大小:", len(m)) // 可能永远无法执行到这里

6. 锁类型与并发规则​

sync.Mutex：排他锁（一刀切） 规则：任何时刻，只有一个 goroutine 能持有锁（无论是读还是写）。 效果：读写操作完全串行化，避免数据竞争，但读操作之间也无法并行（即使只是读取，也会阻塞其他读）。 sync.RWMutex：读写锁（分场景放行） 规则： 读锁（RLock/RUnlock）：可被多个 goroutine 同时持有（读并行）； 写锁（Lock/Unlock）：排他性，持有写锁时，其他 goroutine 既不能读也不能写； 读写互斥：读锁和写锁不能同时存在（读时不能写，写时不能读）。

7. golang中无缓冲和有缓冲的channel区别？

无缓冲Channel是同步通信，发送接收必须同时就绪，像面对面交接，需要频繁切换goroutine。 有缓存Channel是异步通信，有内部队列缓存数据，发送仅在队列满时阻塞，接收仅在队列空时阻塞，像投递到信箱。无缓冲用于精确同步和信号通知，有缓冲用于任务队列、流量控制和性能优化。

func unbufferedExample() {
    ch := make(chan int)  // 无缓冲
    
    go func() {
        fmt.Println("发送前")
        ch <- 42           // 阻塞直到有人接收
        fmt.Println("发送后")
    }()
    
    time.Sleep(time.Second)  // 让发送方等待
    fmt.Println("接收前")
    fmt.Println(<-ch)        // 解除发送方阻塞
    fmt.Println("接收后")
}
// 输出: 发送前 → (等待1秒) → 接收前 → 42 → 接收后 → 发送后

func bufferedExample() {
    ch := make(chan int, 2)  // 容量2
    
    ch <- 1  // 立即成功
    ch <- 2  // 立即成功
    fmt.Println("已发送2个")
    
    ch <- 3  // ❌ 阻塞！缓冲区已满
    fmt.Println("这行不会执行")
}

package main

import (
    "fmt"
    "net/http"
    "strconv"
    "sync/atomic"
    "time"

    "github.com/gin-gonic/gin"
)

// 全局原子计数器（记录总 PV）
var totalPV uint64

// PV 统计中间件
func pvMiddleware() gin.HandlerFunc {
    return func(c *gin.Context) {
        // 原子自增 PV（无锁）
        newPV := atomic.AddUint64(&totalPV, 1)
        // 将 PV 存入上下文（供后续处理使用）
        c.Set("pv", newPV)
        c.Next()
    }
}

func main() {
    r := gin.Default()
    r.Use(pvMiddleware()) // 全局启用 PV 统计

    // 接口：获取实时 PV
    r.GET("/pv", func(c *gin.Context) {
        currentPV := atomic.LoadUint64(&totalPV) // 原子读（避免脏读）
        c.JSON(http.StatusOK, gin.H{"total_pv": currentPV})
    })

    // 接口：模拟业务请求（触发 PV 计数）
    r.GET("/hello", func(c *gin.Context) {
        pv := c.GetUint64("pv") // 从上下文获取当前 PV
        c.JSON(http.StatusOK, gin.H{"message": "hello", "current_pv": pv})
    })

    // 每 10 秒打印一次 PV（模拟监控）
    go func() {
        ticker := time.NewTicker(10 * time.Second)
        defer ticker.Stop()
        for range ticker.C {
            fmt.Printf("当前总 PV: %d\n", atomic.LoadUint64(&totalPV))
        }
    }()

    r.Run(":8080")
}
