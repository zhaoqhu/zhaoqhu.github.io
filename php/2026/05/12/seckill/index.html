<?php
$redis = new Redis();
$redis->connect('127.0.0.1', 6379);

// 1. 商品库存 Key
$stockKey = 'seckill:product:1001:stock';
// 2. 成功抢购的用户集合 Key
$userSetKey = 'seckill:product:1001:users';
// 3. 订单队列 Key (用于异步写入数据库)
$orderListKey = 'seckill:orders:list';

// 初始化库存为 100
$redis->set($stockKey, 100);
// 清空旧数据
$redis->del($userSetKey);
$redis->del($orderListKey);

echo "秒杀初始化完成，库存设置为 100。\n";

第二步：秒杀逻辑(seckill.php)

这是用户点击“抢购”时访问的接口。这是整套系统的核心，没有任何Lua，全是基础命令。

  • 第一层SETNX用户锁，挡住同一用户重复请求。
  • DECR原子扣减，挡住超卖
  • SADD用户去重，挡住一人多单

<?php
$redis = new Redis();
$redis->connect('127.0.0.1', 6379);

$userId = $_GET['user_id'] ?? 'user_' . mt_rand(1000, 9999);
$stockKey = 'seckill:product:1001:stock';
$userSetKey = 'seckill:product:1001:users';
$orderListKey = 'seckill:product:1001:orders'; // 订单队列 Key

// 1. 【前置限制】用 SETNX 挡住 99% 的无效请求（关键！）
$lockKey = 'seckill:lock:' . $userId;
$locked = $redis->set($lockKey, 1, ['nx', 'ex' => 5]);
if (!$locked) {
    die(json_encode(['code' => 400, 'msg' => '请勿重复提交']));
}

// 2. 【库存限制】原子扣减（此时只剩少量请求）
$remainStock = $redis->decr($stockKey);
if ($remainStock < 0) {
    $redis->incr($stockKey); // 回滚
    $redis->del($lockKey);   // 释放锁
    die(json_encode(['code' => 400, 'msg' => '库存不足']));
}

// 3. 【用户限制】Set 去重（双重保险）
$added = $redis->sAdd($userSetKey, $userId);
if (!$added) {
    $redis->incr($stockKey); // 回滚库存
    $redis->del($lockKey);
    die(json_encode(['code' => 400, 'msg' => '重复抢购']));
}

// 4. 异步下单
$redis->lPush('seckill:orders', json_encode(['user_id' => $userId]));
$redis->del($lockKey); // 释放锁

echo json_encode(['code' => 200, 'msg' => '秒杀成功']);

<?php
$redis = new Redis();
$redis->connect('127.0.0.1', 6379);

// 连接数据库（PDO）
$pdo = new PDO('mysql:host=localhost;dbname=test', 'root', 'password');

$orderListKey = 'seckill:product:1001:orders';

echo "订单处理 Worker 启动...\n";

while (true) {
    // 1. 阻塞式取出订单（没有订单就等着，不消耗 CPU）
    $orderJson = $redis->brPop([$orderListKey], 30);
    
    if ($orderJson) {
        $order = json_decode($orderJson[1], true);
        $userId = $order['user_id'];
        
        try {
            // 2. 写入数据库
            $sql = "INSERT INTO orders (user_id, product_id, create_time) VALUES (?, ?, ?)";
            $stmt = $pdo->prepare($sql);
            $stmt->execute([$userId, $order['product_id'], date('Y-m-d H:i:s')]);
            
            echo "[" . date('H:i:s') . "] 订单入库成功: User {$userId}\n";
        } catch (Exception $e) {
            // 3. 入库失败处理（这里简单打印，实际应记录日志或报警）
            echo "[" . date('H:i:s') . "] 订单入库失败: " . $e->getMessage() . "\n";
            // 可选：把失败的订单重新放回队列，或者写入死信队列
        }
    }
}
