<?php
require 'common.php';

$redis = redis_connect();

// 模拟下单
$order_id = 'ORD' . time() . rand(1000, 9999);
$user_id = 1001;
$amount = 99.9;
$delay_seconds = 1800; // 30分钟后关闭

echo "下单成功，订单号：{$order_id}\n";

// 1. 写入 ZSet（时间轮）：score 是执行时间戳
$execute_at = time() + $delay_seconds;
$redis->zAdd('delay_timer', $execute_at, $order_id);

// 2. 写入 Stream（消息仓库）：存储完整订单数据
$message_id = $redis->xAdd(
    'delay_stream',
    '*',  // 自动生成消息ID
    [
        'order_id' => $order_id,
        'data' => create_order($order_id, $user_id, $amount),
        'retry_count' => 0,
        'execute_at' => $execute_at
    ]
);

echo "延迟任务已设置，将在30分钟后执行\n";
echo "ZSet 键：delay_timer\n";
echo "Stream 键：delay_stream\n";
echo "消息ID：{$message_id}\n";

        foreach ($stream_data as $msg_id => $data) {
            if ($data['order_id'] == $order_id) {
                // 3. 将任务推送到 Ready List（就绪队列）
                $redis->lPush('ready_queue', json_encode([
                    'order_id' => $order_id,
                    'message_id' => $msg_id,
                    'data' => $data['data'],
                    'retry_count' => $data['retry_count']
                ]));
                
                echo "任务已推送到就绪队列\n";
                break;
            }
        }
        
        // 4. 从 ZSet 中删除已处理的任务
        $redis->zRem('delay_timer', $order_id);
    }
}

// 5. 休息1秒，避免CPU空转
sleep(1); } ```

<?php
require 'common.php';

$redis = redis_connect();

echo "消费者启动，等待任务...\n";

while (true) {
    // 1. 从 Ready List 阻塞获取任务
    $task_json = $redis->brPop(['ready_queue'], 30);
    
    if ($task_json) {
        $task = json_decode($task_json[1], true);
        $order_id = $task['order_id'];
        $retry_count = $task['retry_count'];
        
        echo "开始处理订单：{$order_id}（第{$retry_count}次尝试）\n";
        
        try {
            // 2. 执行关闭订单逻辑
            $result = close_order($order_id);
            
            if ($result) {
                echo "订单 {$order_id} 关闭成功\n";
                
                // 3. 成功：从 Stream 删除消息
                $redis->xDel('delay_stream', [$task['message_id']]);
                echo "Stream 消息已清理\n";
            } else {
                throw new Exception("关闭订单失败");
            }
        } catch (Exception $e) {
            echo "处理失败：{$e->getMessage()}\n";
            
            // 4. 失败：重试机制
            if ($retry_count < 3) {
                $new_retry = $retry_count + 1;
                $delay = pow(2, $new_retry); // 2秒、4秒、8秒...
                
                echo "将在 {$delay} 秒后重试\n";
                
                // 更新 Stream 中的重试次数
                $redis->xAdd('delay_stream', $task['message_id'], [
                    'order_id' => $order_id,
                    'data' => $task['data'],
                    'retry_count' => $new_retry,
                    'execute_at' => time() + $delay
                ]);
                
                // 重新加入 ZSet（延迟重试）
                $redis->zAdd('delay_timer', time() + $delay, $order_id);
            } else {
                // 5. 超过最大重试次数，进入死信队列
                echo "任务 {$order_id} 进入死信队列\n";
                $redis->lPush('dead_letter_queue', json_encode([
                    'order_id' => $order_id,
                    'error' => $e->getMessage(),
                    'final_retry' => $retry_count
                ]));
                
                // 从 Stream 删除
                $redis->xDel('delay_stream', [$task['message_id']]);
            }
        }
    }
}

// 模拟关闭订单的函数
function close_order($order_id) {
    // 这里写实际的关闭订单逻辑
    // 比如更新数据库订单状态为已关闭
    
    // 模拟 90% 成功率
    $success = rand(1, 10) <= 9;
    
    if ($success) {
        // 更新数据库...
        return true;
    } else {
        return false;
    }
}

<?php
require 'common.php';

$redis = redis_connect();

echo "=== 延迟队列监控 ===\n";
echo "时间：" . date('Y-m-d H:i:s') . "\n\n";

// 1. 待处理任务（ZSet）
$pending = $redis->zCard('delay_timer');
echo "待处理任务数：{$pending}\n";

// 2. 就绪队列（List）
$ready = $redis->lLen('ready_queue');
echo "就绪队列任务数：{$ready}\n";

// 3. Stream 消息数
$stream_info = $redis->xInfo('STREAM', 'delay_stream');
echo "Stream 消息数：{$stream_info['length']}\n";

// 4. 死信队列
$dead = $redis->lLen('dead_letter_queue');
echo "死信队列任务数：{$dead}\n\n";

// 5. 查看具体内容
echo "=== 待处理任务（前5个）===\n";
$tasks = $redis->zRange('delay_timer', 0, 4, true);
foreach ($tasks as $order_id => $timestamp) {
    echo "订单：{$order_id}，执行时间：" . date('H:i:s', $timestamp) . "\n";
}

echo "\n=== 死信队列（前3个）===\n";
$dead_tasks = $redis->lRange('dead_letter_queue', 0, 2);
foreach ($dead_tasks as $task) {
    $data = json_decode($task, true);
    echo "订单：{$data['order_id']}，错误：{$data['error']}\n";
}
